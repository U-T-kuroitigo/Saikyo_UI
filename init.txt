# ===== 起動テスト手順 =====

# 1) .env を用意（.env.example をコピーして値を埋める）
cp .env.example .env

# 2) ビルド & 起動
docker compose up -d --build

# 3) コンテナの状態確認
docker compose ps

# 4) healthcheck 通過確認（app, db が healthy になるか確認）
# Windows PowerShell では jq が使えないので以下のように確認
#   docker inspect saikyo_api | Select-String '"Status"'
# Linux/Mac の場合は jq が利用可能
#   docker inspect --format='{{json .State.Health}}' saikyo_api | jq

# 5) API動作確認
curl http://localhost:8080/health
# => HTTP/1.1 200 OK が返ること

# 6) 問題があればログを確認
docker compose logs app
docker compose logs db


# ===== プロジェクトファイル全てに自動整形 =====
# Go以外のファイル
npx prettier --write .
# Goファイル
go fmt ./...
